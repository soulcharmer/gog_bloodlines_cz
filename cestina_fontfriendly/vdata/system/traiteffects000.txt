//
//	Trait-Effects are things that modify a character's Stats (Attributes/Abilities/etc.) or related game-system rules,
// Such as Stat Buy/Sell Costs.
//
//	There are 5 Categories: Clan, History, Discipline, Numina and Item, used to separate the different types of effects.
//	Inside each category, there are multiple Trait-Effect Groups, that may contain any number of Trait-Effects.  So a
// given Trait-Effect Group may actually do multiple things.  A Trait-Effect Group is what you actually map to via its
// "InternalName".
//	Each Trait-Effect is specified as in:
//
//	TraitEffect
//	{
//		"Trait"	-- The Trait (Attribute/Ability/etc.) to modify
//		"Modify"	-- How to modify the Trait:
//					"+" or "-"	-- Add or subtract to the base value
//					"*"		-- Multiply the base value
//					"/"		-- Divide the base value
//					"Max"		-- The Maximum possible value is this
//					"Min"		-- The Minimum possible value is this
//					"%"		--
//					"Value"	-- Ignore the base value.  Use this instead.
//					"Cost"	-- Use this cost structure instead of the default.
//					"BloodCost"	-- Modify the BloodCost to activate this (discipline) by a given value (+/-).
//					"PoÂkození"	-- Modify the PoÂkození from this (discipline) by a given %.
//					"Trvání"	-- Modify the Trvání of this (discipline) by a given %.
//	}
//
//	Note that for changing Buy/Sell costs, you can skip the "Modify" field, but you need to add the Cost structure:
//
//	TraitEffect
//	{
//		"Trait"	-- The Trait (Attribute/Ability/etc.) to modify
//		Costs
//		{
//			"New"		"5"				-- Optional, only needed if you can sell down to "0".  This is the initial cost to buy (from 0->1).
//			"Raise"	"Current_Rating * 1"	-- This is the cost of the Trait
//		}
//	}
//
//
